Perform Elbow Method to find Optimal No. of Clusters¶

In [78]:
from sklearn.cluster import KMeans
wcss = []
In [79]:
for i in range(1, 11):
    kmeans = KMeans(n_clusters=i, init='k-means++', random_state=0)
    kmeans.fit(X)
    wcss.append(kmeans.inertia_)
In [80]:

plt.show()
